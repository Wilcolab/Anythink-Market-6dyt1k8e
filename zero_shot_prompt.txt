import re

def to_snake_case(text):
    # Replace spaces and hyphens with underscores
    text = re.sub(r'[\s\-]+', '_', text)
    # Insert underscores before capitals (except the first char)
    text = re.sub(r'([a-z])([A-Z])', r'\1_\2', text)
    # Lowercase everything and remove any non-alphanumeric/underscore
    text = re.sub(r'[^a-zA-Z0-9_]', '', text)
    return text.lower()

# Example usage
print(to_snake_case("Convert This-Text to snakeCase!"))
# Output: convert_this_text_to_snake_case
